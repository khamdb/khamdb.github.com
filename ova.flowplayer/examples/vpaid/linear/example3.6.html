Example 3.4 - Deferred VPAID pre-roll ad call


The configuration for this example is: 


flowplayer("a.player614x345", "", {
    plugins: {
        controls: {
            autoHide: "never"
        },
        
        ova: {
            url: "",

            "delayAdRequestUntilPlay": true,
 
            "shows": {
                "baseURL": "",
                "streams": [
                    {  
                        "file":"", 
                        "duration":"00:00:20"
                    }                    
                ]
            },
            
            "ads": {
               "schedule": [
                  {
                     "position": "pre-roll",
                     "tag": ""
                  }
               ]
            },

            "debug": {
              "levels": ""
            }
        }
    }
});
