Example 3.15.00 - Forcing OVA to Stop a VPAID ad given a specific log message


The configuration for this example is: 


function onVPAIDAdLog(ad, message) {
	console.log(">> onVPAIDAdLog() fired - '" + message + "'");
}

flowplayer("a.player614x345", "", {
    plugins: {
        controls: {
            autoHide: "never"
        },
        
        ova: {
            url: "",
 
            "shows": {
                "baseURL": "",
                "streams": [
                    {  
                        "file":"", 
                        "duration":"00:00:20"
                    }                    
                ]
            },
            
            "ads": {
               "vpaid": {
                  "terminateOnLogMessage": "__single-quote__errorEvent: No ads__single-quote__"
               },
               "schedule": [
                  {
                     "position": "pre-roll",
                     "tag": "../../../dist/templates/vpaid/vast2-vpaid1-linear-log-terminate.xml"
                  }
               ]
            },

            "debug": {
              "levels": ""
            }
        }
    }
});
