Example 6 - 468x60 not scalable but ad slot scaling enabled


The configuration for this example is: 


flowplayer("example", "", {
    playlist: [
       {
           url: "",
           duration: 20
       }
    ],

    plugins: {	    
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            "url": "",

            "ads": {
                "schedule": [
                    {
                       "region": "auto:bottom",
                       "startTime": "00:00:03",
                       "duration": "recommended:15",
                       "enableScaling": true,
                       "tag": "../../../../../dist/templates/overlays/scaling/vast2-swf-not-scalable-but-fits.xml"
                    }
                ]
            }
            
            "debug": {
                "levels": ""
            }
        }
    }
});
