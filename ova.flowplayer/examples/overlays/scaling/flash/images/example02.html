Example 2 - 728x90 image scaled to declared width

Scalable image overlay that is automatically scaled to a declared ad slot width and height


The configuration for this example is: 


flowplayer("example", "", {
    playlist: [
       {
           url: "",
           duration: 20
       }
    ],

    plugins: {	    
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            "url": "",

            "ads": {
                "schedule": [
                    {
                       "region": "auto:bottom",
                       "startTime": "00:00:03",
                       "duration": "recommended:15",
                       "enableScaling": true,
                       "width": 364,
                       "height": 45, 
                       "alwaysMatch": true,                       
                       "tag": "../../../../../dist/templates/overlays/scaling/vast2-image-scalable.xml"
                    }
                ]
            }
            
            "debug": {
                "levels": ""
            }
        }
    }
});
