Example 6 - VPAID Overlays (expandable VAST sizing)

The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
       {
           url: "",
           duration: 60
       }
    ],
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "ads": {
                "schedule": [
                    {    
                      "region": "auto:bottom",
                      "startTime": "00:00:03",
                      "duration": 30,
                      "tag": "http://www.adotube.com/..videoURL=[VIDEO_FILE_URL]"
                    }
                ]
            },

            "debug": {
                "levels": ""
            }
        }
    }
});
