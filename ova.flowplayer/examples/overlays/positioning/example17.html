Example 17 - Explicitly positioning all HTML5 enabled regions as "overlays"

The configuration for this example is: 


flowplayer("a.example", { src: "", wmode: "opaque" }, {
    playlist: [
       {
           url: "",
           duration: 40
       }
    ],
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "ads": {
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "",
                       "allowAdRepetition": true
                    }
                ],
                "overlays": {
                    "overlay": { "html5": true },
                    "regions": {
                       "flash": false,
                       "html5": true
                    }
                },
                "schedule": [
                    {    
                      "zone": "28",
                      "width": 450,
                      "height": 50,
                      "startTime": "00:00:03",
                      "duration": 10
                    }
                ]
            },

            "debug": {
                "levels": ""
            }
        }
    }
});
