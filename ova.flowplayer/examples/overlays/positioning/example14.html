Example 14 - Positioning all HTML5 enabled regions across all ad slots as "non-overlays"

The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
       {
           url: "",
           duration: 40
       }
    ],
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "ads": {
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "",
                       "allowAdRepetition": true
                    }
                ],
                "overlays": {
                    "overlay": { "html5": false }
                },
                "schedule": [
                    {    
                      "zone": "28",
                      "width": 450,
                      "height": 50,
                      "startTime": "00:00:03",
                      "duration": 10,
                      "regions": {
                          "flash": false,
                          "html5": true
                      }
                    }
                ]
            },

            "debug": {
                "levels": ""
            }
        }
    }
});
