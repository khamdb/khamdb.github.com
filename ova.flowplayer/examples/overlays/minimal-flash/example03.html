Example 3 - Default "auto:bottom" position, ad type based content selection

The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
       {
           url: "",
           duration: 40
       }
    ],
    plugins: {
        controls: {
            autoHide: "always"
        },
        rtmp: {
            url: ""
        },
        rtmpInstream: {
            url: ""
        },

        ova: {
            url: "",

            "ads": {
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "",
                       "allowAdRepetition": true
                    }
                ],
                "schedule": [
                    {    
                      "zone": "41",
                      "acceptedAdTypes": [ "image" ],
                      "startTime": "00:00:03",
                      "duration": 10
                    }
                ]
            },

            "debug": {
                "levels": ""
            }
        }
    }
});
