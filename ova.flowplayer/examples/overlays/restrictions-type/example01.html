Example 1 - Restricting content selection based on ad type across all ad slots

The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
       {
           url: "",
           duration: 40
       }
    ],
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "ads": {
                "overlays": {
                   "acceptedAdTypes": [ "image" ]
                },
                "schedule": [
                    {    
                      "zone": "41",
                      "startTime": "00:00:03",
                      "duration": 10,
                      "tag": "../../../dist/templates/overlays/vast2-multi-types.xml"
                    }
                ]
            },

            "debug": {
                "levels": ""
            }
        }
    }
});
