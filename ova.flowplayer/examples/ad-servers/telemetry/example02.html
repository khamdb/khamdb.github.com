Example 2 - VAST2 VPAID (via wrapper)

This example illustrates a simple VPAID non-linear example.


The configuration for this example is: 


flowplayer("a.example", "", {
    plugins: {
        controls: {
        	autoHide: "never"
        },

        ova: {
            url: "",
 
            "shows": {
                "baseURL": "",
                "streams": [
                    {  
                        "file":"", 
                        "duration":"00:00:20"
                    }                    
                ]
            },
            
            "ads": {
               "schedule": [
                   {
                       "position": "pre-roll",
                       "tag": "../../../dist/templates/ad-servers/telemetry/telemetry-wrapper-vast2.xml"
                   }
               ]
            },

            "debug": {
              "levels": ""
            }
        }
    }
});
