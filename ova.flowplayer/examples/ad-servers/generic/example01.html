Example 1 (HTTP) - A VAST 1.0 pre-roll ad only



The configuration for this example is: 


flowplayer("a.example", "", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "debug": {
                "levels": ""
            },

            "ads": {
                "pauseOnClickThrough": true,
                "servers": [
                   {
                       "type": "direct",
                       "apiAddress": "?
                                      script=bannerTypeHtml:vastInlineBannerTypeHtml:
                                      vastInlineHtml__amp__zones=pre-roll0-0%3D5__amp__
                                      nz=1__amp__source=__amp__r=R0.22563580377027392__amp__
                                      block=1__amp__format=vast__amp__charset=UTF-8"
                   }
                ],
                "schedule": [
                    {
                       "zone": "5",
                       "position": "pre-roll"
                    }
                 ]
             }
         }
     }
});
