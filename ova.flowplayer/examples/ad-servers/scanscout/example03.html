Example 1 - A VAST 1.0 Linear (Pre-Roll) with Companion(s)



The configuration for this example is: 


flowplayer("example", "", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            url: "",

            "shows": {
              "streams": [
                   {
                     "file": "",
                     "duration":"00:00:20"
                   }
              ]
            },

            "ads": {
                "pauseOnClickThrough": true,
                "schedule": [
                    {
                       "position": "auto:bottom",
                       "duration": "recommended:10",
                       "startTime": "00:00:05",
                       "server": {
                          "type": "direct",
                          "tag": "",
                       }
                    }
                ]
            }

            "debug": {
                "levels": ""
            },

        }
    }
});
