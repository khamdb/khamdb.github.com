Example 1 (HTTP) - A VAST 1.0 pre-roll ad only



The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
       {
           url: OVA_HTTP_SHOW_STREAM_1,
           duration: 20
       }
    ],	
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "debug": {
                "levels": ""
            },

            "ads": {
                "pauseOnClickThrough": true,
                "servers": [
                   {
                       "type": "AdOcean",
                       "apiAddress": "http://myao.adocean.pl/ad.xml?id=3vSgBYmXlxZyWPJifZWPdpr7obNjbEcaLYfaLQI6nRb.y7",
                       "oneAdPerRequest": false
                   }
                ],
                "schedule": [
                    {
                       "position": "pre-roll",
                       "repeat": 2
                    },
                    {
                      "position": "mid-roll",
                      "startTime": "00:00:08"
                    },
                    {
                      "position": "mid-roll",
                      "startTime": "00:00:15"
                    },
                    {
                       "position": "post-roll"
                    }
                 ]
             }
         }
     }
});
