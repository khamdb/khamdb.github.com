Example 1- VAST 2.0 Non-Linear

Overlay starts 3 seconds in, runs for a minimum of 10 seconds.

300x250 Companion
728x90 Companion

The configuration for this example is: 


flowplayer("example", "", {
    plugins: {	    
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",
            
            "debug": {
                "levels": ""
            },

            "ads": {
                "pauseOnClickThrough": true,
                "companions": {
                    "regions": [
                       {  
                          "id":"companion-300x250", 
                          "width":"300", 
                          "height":"250"
                       },
                       { 
                          "id":"companion-728x90", 
                          "width":"728", 
                          "height":"90" 
                       }
                    ]
                },
                "schedule": [
                     {
                        "position": "auto",
                        "duration": "recommended:15",
                        "startTime": "00:00:03",
                        "server": {
                           "type": "direct",
     	                   "tag": "http://demo.tremormedia.com/proddev/vast/vast_inline_nonlinear.xml"
                        }
                     }
                ]
            }
        }
    }
});
