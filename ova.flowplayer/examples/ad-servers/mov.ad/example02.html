Example 2 - VAST 2.0 Linear (Pre-Roll)

300x250 Companion

The configuration for this example is: 


flowplayer("example", "", {
    plugins: {	    
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",
            
            "debug": {
                "levels": ""
            },

            "ads": {
                "pauseOnClickThrough": true,
                "companions": [
                     {  
                        "id":"companion-300x250", 
                        "width":"300", 
                        "height":"250"
                     }
                ],
                "schedule": [
                     {
                        "position": "pre-roll"
                        "server": {
                           "type": "direct",
                           "tag": "http://v.movad.de/stream.ad?cid=101224"
                        }
                     }
                ]
            }
        }
    }
});
