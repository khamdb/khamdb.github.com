Example 1- VAST 1.0 Linear (Pre-Roll)


The configuration for this example is: 


flowplayer("example", "", {
    plugins: {	    
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",
            
            "debug": {
                "levels": ""
            },

            "ads": {
                "pauseOnClickThrough": true,
                "schedule": [
                     {
                        "position": "pre-roll"
                        "server": {
                           "type": "direct",
                           "tag": "http://v.movad.de/stream.ad?cid=101226"
                        }
                     }
                ]
            }
        }
    }
});
