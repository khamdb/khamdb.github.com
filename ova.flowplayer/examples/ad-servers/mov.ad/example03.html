Example 3 - VAST 2 VPAID Linear Ad


The configuration for this example is: 


flowplayer("example", "", {
    plugins: {	    
        controls: {
            autoHide: "never"
        },

        ova: {
            url: "",
            
            "debug": {
                "levels": ""
            },

            "ads": {
                "pauseOnClickThrough": true,
                "companions": [
	                  { "id":"companion-300x250", "width":"300", "height":"250" }
                ],
                "schedule": [
                    {
                       "position": "pre-roll",
                       "server": {
                           "type": "direct",
      	                   "tag": ""
                       }
                    }
                ]
            }
        }
    }
});
