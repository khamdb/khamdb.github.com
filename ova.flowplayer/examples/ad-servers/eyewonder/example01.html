Example 1 - Linear VPAID (Pre-roll)



The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
        {
           url: "",
           duration: 20
        }
    ],

    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            "url": "",

            "debug": {
                "levels": ""
            },

            "ads": {
                "schedule": [
                   {
                      "position": "pre-roll",
                      "tag": ""
                   }
                ]
            }
        }
     }
});
