Test 06 - 1 stream with pre, mid and post rolls

1 stream with a pre-roll, an overlay 5 seconds in for 5 seconds, a mid-roll at 10 seconds into the main stream followed by a post-roll at the end = 4 streams


The configuration for this example is: 


flowplayer("a.example", "", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "shows": {
              "baseURL": "",
              "streams": [
                   { "file":"" }
              ]
            },

            "ads": {
              "servers": [
                  {
                     "type": "OpenX",
                     "apiAddress": "",
                     "allowAdRepetition": true
                  }
              ],
              "schedule": [
                  {
                      "zone": "19",
                      "position": "pre-roll"
                  },
                  {
                      "zone": "19",
                      "position": "mid-roll",
                      "startTime": "00:00:10"
                  },
                  {
                      "zone": "19",
                      "position": "post-roll"
                  },
                  {
                      "zone": "30",
                      "startTime": "00:00:05",
                      "duration": 5,
                      "width": 450,
                      "height": 50
                  }                  
              ]
            },

            "debug": {
              "levels": ""
            }
        }
     }
});
