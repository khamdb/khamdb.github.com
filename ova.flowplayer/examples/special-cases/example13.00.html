Example 13 - Security Exception - Standard Playlist

This example tests what happens when a bad VAST response is returned. The ad schedule should be skipped and the main stream(s) played.


The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
        {  
            url: "",
            duration: 10
        },
        {  
            url: "",
            duration: 11
        },
    ],

    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "debug": {
                "levels": ""
            },

            "ads": {
                "servers": [
                    {
                       "type": "direct",
                       "apiAddress": "http://ads.example.com/this-will-fail.html"
                    }
                ],

                "schedule": [
                    {
                       "zone": "5",
                       "position": "pre-roll",
                       "applyToParts": [ 0 ]
                    },
                    {
                       "zone": "5",
                       "position": "post-roll"
                    }
                ]
           }
     }
});
