Example 20 - Supporting Audio Ads

The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
        { 
            url: "", 
            duration: 30,
            provider: "audio" 
        }
    ],
    
    plugins: {
        controls: {
            fullscreen: false,
            autoHide: false,
            height: 30
        },

        audio: {
            url: ""
        },

        ova: {
            url: "",

            "providers": {
       	       "http": "audio"
            },

            "ads": {
                "acceptedLinearAdMimeTypes": [ "audio/x-mpeg-3" ],
                "filterOnLinearAdMimeTypes": true,
                "schedule": [
                    {
                      "position": "pre-roll",
                      "tag": ""
                    }
                ]
            },

            "debug": {
                "levels": ""
            }
        }
    }
});
