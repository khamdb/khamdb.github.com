Example 02 - Survey with Restore

A 300x250 SURVEY GOES HERE

The configuration for this example is: 


flowplayer("a.example", "", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "shows": {
                "baseURL": "",
                "streams": [
                    { "file":"", "duration":"00:00:20" }
                ]
            },

            "ads": {
               "survey": {
                   "id": "survey",
                   "restore": true
               },
               "schedule": [
                   {
                      "position": "pre-roll",
                      "tag": "http://static.openvideoads.org/tags/survey-vast.xml"
                   }
               ]
            },

            "debug": {
                "levels": ""
            }
        }
    }
});
