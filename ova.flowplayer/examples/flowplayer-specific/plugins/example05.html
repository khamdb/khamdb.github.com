Example 5 - Using OVA with the Sharing plugin

The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
       {
           url: OVA_HTTP_SHOW_STREAM_1,
           duration: 10,
           pageUrl: "http://www.openvideoads.org"
       }
    ],

    plugins: {
        sharing: {
            url: "",
        },

        ova: {
            url: "",
            
            "playOnce": true,
            
            "ads": {
              "servers": [
                   {
                      "type": "OpenX",
                      "apiAddress": "",
                      "allowAdRepetition": true
                   }
              ],
              "schedule": [
                  {
                     "zone": "5",
                     "position": "pre-roll"
                  }
              ]
            },

            "debug": {
              "levels": ""
            }
        }
    }
});
