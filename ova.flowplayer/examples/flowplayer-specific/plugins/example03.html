Example 19 - Using OVA with a proxy plugin (e.g. Akamai)

The configuration for this example is: 


flowplayer("a.example", "", {
    log: { level: 'debug', filter: 'org.flowplayer.akamai.*' },
    
    clip: {  
        provider: 'rtmp',
    },
    
    plugins: {
        rtmp: {
            url: "",
            proxyType: "best"
        },

        akamai: {
            url: "" 
        },

        ova: {
            url: "",
            
            "ads": {
              "enableProxies": true, 
              "schedule": [
                  {
                     "position": "pre-roll",
                     "tag": "http://ad.doubleclick.net/adx/ibs.hou.interactive/local;
                             kw=news+19032176+300k+clienttrack+playerpage+preroll;
                             dcmt=text/xml;sz=2x1;pool=hou;ord=1300808587430.asx"
                  }
              ]
            },

            "debug": {
              "levels": ""
            }
        }
    }
});
