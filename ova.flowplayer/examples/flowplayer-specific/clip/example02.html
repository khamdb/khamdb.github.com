Example 2 - Applying Clip Properties to Ad Clips

This example illustrates how to apply clip properties to ad clips.

The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
       {
           url: "http://www.vtelevision.es/default/2010/04/21/00311271849966888398784/Video/Crimen Teixeiro.flv"
       }
    ],
    
    clip: {
    	eventCategory: "video"
    },
    
    plugins: {
        rtmp: {
            url: "",
            netConnectionUrl: ""
        },

        controls: {
            autoHide: "always",
            playlist: true
        },

        ova: {
            url: "",

            "player": {
            	"applyCommonClipProperties": true
            },
            
            "ads": {
              "servers": [
                   {
                      "type": "OpenX",
                      "apiAddress": ""
                   }
              ],
              "schedule": [
                  {
                     "zone": "5",
                     "position": "pre-roll"
                  }
              ]
            },

            "debug": {
              "levels": ""
            }
        }
    }
});
