Example 2.18.00 - Pausing/Resuming the active ad - Linear ad stream


  • Pause
  • Resume

The configuration for this example is: 


<script>
flowplayer("a.example", "", {
    playlist: [
        {
            url: "",
            duration: 30
        }
    ],

    plugins: {	    
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "ads": {
                 "servers": [
                     {
                          "type": "OpenX",
                          "apiAddress": "",
                     }
                 ],
                 "notice": { "textStyle": "smalltext" },
                 "schedule": [
                       {
                           "zone": "5",
                           "position": "pre-roll"
                       }
                 ]
            },

            "debug": {
                 "levels": ""
            }
        }
    }
});
</script>

<ul>
  <li><a href="#" onclick="$f().getPlugin("ova").pause();">
    Pause
  </a></li>
  <li><a href="#" onclick="$f().getPlugin("ova").resume();">
    Resume
  </a></li>
</ul>
