Example 6 - VAST FLV RTMP Media Files Without File Markers - No Streamers Specified

The configuration for this example is: 


flowplayer("a.example", "", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        rtmp: {
            url: ""
        },

        ova: {
            url: "",

            "shows": {
                "baseURL": "",
                "streams": [ { "file":"", "duration":"00:00:34" } ]
            },

            "ads": {
                "schedule": [
                    {   
                      "position": "pre-roll",
                      "tag": ""
                    }
                ]
            },

            "debug": {
                "levels": ""
            }
        }
    }
});
