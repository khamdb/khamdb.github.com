Example 5.00.00 - Selective enabling - all linear ad types

The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
    	{
    		url: "",
    		duration: 30
    	}
    ],
    
    plugins: {
        controls: {
            autoHide: "never"
        },

        ova: {
              url: "",

              "player": {
                 "modes": {
                    "linear": {
                       "controls": { 
                          "enablePlay": true,
                          "enablePause": true,
                          "enablePlaylist": false,
                          "enableFullscreen": false,
                          "enableTime": false,
                          "enableMute": false,
                          "enableVolume": false
                       }
                    }
                 }
              },
              
              "ads": {
                 "servers": [
                     {
                        "type": "OpenX",
                        "apiAddress": ""
                     }
                 ],
                 "schedule": [
                     {
                        "zone": "5",
                        "position": "pre-roll"
                     },
                     {
                        "zone": "5",
                        "position": "post-roll"
                     }
                 ]
              },

              "debug": {
                 "levels": ""
              }
        }
    }
});
