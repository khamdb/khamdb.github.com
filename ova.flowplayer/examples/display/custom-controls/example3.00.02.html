Example 3.00.02 - Showing the control bar during linear ad playback - VPAID ads

The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
    	{
    		url: "",
    		duration: 30
    	}
    ],
    
    plugins: {
        controls: {
            autoHide: "never"
        },

        ova: {
              url: "",
            
              "player": {
                 "modes": {
                    "linear": {
                       "controls": { 
                           "vpaid": { "visible": true }
                       }
                    }
                 }
              },
              
              "ads": {
                 "schedule": [
                     {
                        "position": "pre-roll",
                        "tag": ""
                     }
                 ]
              },
              
              "debug": {
                 "levels": ""
              }
        }
    }
});
