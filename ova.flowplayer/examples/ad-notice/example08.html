Example 8 - Turning off the Click-Through Message

You can modify the font style, size and whether or not the message is shown.


The configuration for this example is: 


flowplayer("a.example", "", {
    plugins: {
        controls: {
            autoHide: "always"
        },
        rtmp: {
            url: ""
        },

        ova: {
            url: "",

            "debug": {
                "levels": ""
            },

            "ads": {
                "clickSign": {
                    "enabled": false
                },
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "",
                       "customProperties": { 
                            "target": "category=food" 
                       }
                    }
                ],
                "schedule": [
                    {
                       "zone": "1",
                       "position": "pre-roll"
                    }
                ]
            }
        }
    }
});
