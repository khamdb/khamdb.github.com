Example 4 - Modifying the Linear Ad Clickthrough Message

You can modify the font style, size and whether or not the message is shown.


The configuration for this example is: 


flowplayer("a.example", "", {
    plugins: {
        controls: {
            autoHide: "always"
        },
        rtmp: {
            url: ""
        },

        ova: {
            url: "",

            "debug": {
                "levels": ""
            },

            "ads": {
                "pauseOnClickThrough": true,
                "clickSign": {
                    "enabled": true, 
                    "verticalAlign": "center",
                    "horizontalAlign": "center",
                    "width": 150,
                    "height": 32,
                    "opacity": 0.5,
                    "borderRadius": 20,
                    "backgroundColor": "#000000",
                    "style": ".smalltext { font-size:12; }",
                    "html": "<p class=\"smalltext\" align=\"center\">CLICK ME!</p>",
                    "scaleRate": 0.75
                },
                "notice": { "show": true },
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "",
                       "customProperties": { 
                            "target": "category=food" 
                       }
                    }
                ],
                "schedule": [
                    {
                       "zone": "1",
                       "position": "pre-roll"
                    }
                ]
            }
        }
    }
});
