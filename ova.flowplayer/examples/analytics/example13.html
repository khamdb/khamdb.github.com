Example 13 - Turning on the Ad Tag param in the tracking URLs



The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
        {
            url: OVA_HTTP_SHOW_STREAM_1,
            duration: 30
        }
    ],

    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "analytics": {
                "google": {
	                "custom": {
                        "accountId": "UA-10158120-1",
                        "trackAdTags": true,
                        "impressions": {
                            "enable": true
                        }
                    }
                }
            },

            "debug": {
                "levels": ""
            },

            "ads": {
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "",
                    }
                ],
                "schedule": [
                    {
                       "zone": "5",
                       "position": "pre-roll"
                    }
                ]
            }
        }
    }
});
