Example 8 - A mid-roll on a playlist defined stream with a start time and duration



The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
       {
          "url": "",
          "duration": 30,
          "start": 10,
          "provider": "pseudo"
       }
    ],

    plugins: {
        controls: {
            autoHide: "always"
        },

        pseudo: {
            url: ""
        },
        pseudoInstream: {
            url: ""
        },

        ova: {
            url: "",

            "debug": {
                "levels": ""
            },

            "ads": {
                "playOnce": false,
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": ""
                    }
                ],
                "schedule": [
                    {
                       "zone": "5",
                       "position": "mid-roll",
                       "startTime": "00:00:10"
                    }
                ]
            }
        }
    }
});
