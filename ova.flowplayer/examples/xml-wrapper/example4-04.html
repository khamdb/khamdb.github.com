Example 4-04 - Chaining a VAST1 wrapper to a VAST2 wrapper to a VAST2 ad



The configuration for this example is: 


flowplayer("example", "", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "shows": {
              "streams": [
                   {
                     "file": "",
                     "duration":"00:00:10"
                   }
              ]
            },

            "ads": {
                "pauseOnClickThrough": true,
                "companions": { 
                	"regions": [ { id: "companion-1", width: 300, height: 250 } ]
                },
                "schedule": [
                    {
                       "position": "pre-roll",
                       "server": {
                          "type": "direct",
                          "tag": "",
                       }
                    }
                ]
            }

            "debug": {
                "levels": ""
            }
        }
    }
});
