Example 1-02 - VAST 2 to VAST 2 linear ad

This example illustrates a VAST2 wrapper in action - a single layer of wrapper without ad templates.



The configuration for this example is: 


flowplayer("example", "", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "shows": {
              "streams": [
                   {
                     "file": "",
                     "duration":"00:00:10"
                   }
              ]
            },

            "ads": {
                "pauseOnClickThrough": true,
                "schedule": [
                    {
                       "position": "pre-roll",
                       "server": {
                          "type": "direct",
                          "tag": "",
                       }
                    }
                ]
            }

            "debug": {
                "levels": ""
            }
        }
    }
});
