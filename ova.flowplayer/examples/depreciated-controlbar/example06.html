Example 6 - Stopping OVA from enabling/disabling the control bar

This example illustrates how to stop OVA from enabling/disabling the control bar.

The configuration for this example is: 


flowplayer("a.example", "", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "shows": {
                 "streams": [
                      {
                          "file": "",
                          "duration":"00:00:30"
                      }
                 ]
            },

            "ads": {
                 "controls": {
                     "manage": false
                 },
                 "servers": [
                     {
                           "type": "OpenX",
                           "apiAddress": "",
                     }
                 ],
                 "schedule": [
                      {
                           "zone": "5",
                           "position": "pre-roll"
                      }
                  ],
              },

              "debug": {
                  "levels": "",
                  "debugger": "firebug"
              }
         }
    }
});
