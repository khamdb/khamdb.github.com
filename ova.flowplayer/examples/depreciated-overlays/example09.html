Example 9 - Enabling the Cross Hair Close Button


The configuration for this example is: 


flowplayer("example", "", {
    plugins: {	    
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            "url": "",

            "shows": {
                "baseURL": "",
                "streams": [ { "file":"", "duration":"00:00:34" } ]
            },

            "overlays": {
            	"closeButton": { "type": "crosshair", "color": "#FFFFFF" }
            },

            "ads": {
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": ""
                    }
                ],
                "pauseOnClickThrough": true,
                "schedule": [
                    {
                       "zone": "28",
                       "startTime": "00:00:05",
                       "position": "auto:center",
                       "duration": "recommended:10"
                    }
                ]
            }
            
            "debug": {
                "debugger": "firebug",
                "levels": ""
            }
        }
    }
});
