Example 18 - Putting Overlays into Debug Mode


The configuration for this example is: 


flowplayer("example", "", {
    playlist: [
       {
           url: "",
           duration: 20
       }
    ],

    plugins: {	    
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            "url": "",

            "overlays": {
                "debug": true
            },
            
            "ads": {
                "schedule": [
                    {
                       "position": "auto:bottom",
                       "startTime": "00:00:05",
                       "duration": "recommended:10",
                       "tag": ""
                    }
                ]
            }
            
            "debug": {
                "levels": ""
            }
        }
    }
});
