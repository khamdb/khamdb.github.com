Example 15 - Supporting Extension Tags


The configuration for this example is: 


flowplayer("example", "", {
    playlist: [
       {
           url: OVA_HTTP_SHOW_STREAM_1,
           duration: 20
       }
    ],

    plugins: {	    
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            "url": "",

            "ads": {
                "schedule": [
                    {
                       "position": "auto:center",
                       "startTime": "00:00:05",
                       "duration": "recommended:10",
                       "acceptedAdTypes": [ "image" ]
                       "tag": ""
                    }
                ]
            }
            
            "debug": {
                "levels": ""
            }
        }
    }
});
