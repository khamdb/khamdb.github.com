Example 20 - Forcing mid-roll playback when seeking during a show clip

The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
       {
           url: "",
           duration: 60
       }
    ],
    
    plugins: {
        ova: {
            url: "",

            "ads": {
            	"enforceMidRollPlayback": true,
                "servers": [
                    {
                      "type": "OpenX",
                      "apiAddress": ""
                    }
                ],
                "schedule": [
                    {
                      "zone": "6",
                      "position": "pre-roll"
                    },                
                    {
                      "zone": "5",
                      "position": "mid-roll",
                      "startTime": "00:00:15"
                    }
                ]
            },

            "debug": {
                "levels": ""
            }
        }
    }
});
