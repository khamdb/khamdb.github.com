Example 19-1 - Repeating Ad positions at specified intervals - overlays

The configuration for this example is: 


flowplayer("a.example", "", {
   plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "shows": {
                "baseURL": "",
                "streams": [
                    { "file":"", "duration":"00:01:00" }
                ]
            },

            "ads": {
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "",
                       "allowAdRepetition": true
                    }
                ],
                "schedule": [
                    {
                       "zone": "30",
                       "position": "auto:bottom",
                       "startTime": "00:00:10",
                       "duration": 5,
                       "repeat": 3,
                       "interval": 10
                    }
                ]
            },

            "debug": {
                "levels": ""
            }
        }
    }
});
