Example 2 - Customising the "skip ad" button image


The configuration for this example is: 


flowplayer("a.example", "", {
    plugins: {
        controls: {
            autoHide: "never"
        },

        ova: {
            url: "",

            "shows": {
                "streams": [
                    {
                        "file": "",
                        "duration":"00:00:30"
                    }
                ]
            },

            "debug": {
                "levels": "",
            },

            "ads": {
                "skipAd": {
                    "enabled": true,
                    "image": "",
                    "width": 20,
                    "height": 15
                },
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "",
                    }
                ],
                "schedule": [
                    {
                       "zone": "5",
                       "position": "pre-roll"
                    }
                ]
            }
        }
    }
});
